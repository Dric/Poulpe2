 85     /**
 86      * Sauvegarde les informations du profil
 87      *
 88      * @return bool
 89      */
 90     protected function saveUserProfile(){
 91         $req = $this->postedData;
 92         $avatar = $req['avatar'];
 93         // Si un fichier est chargé, alors l'utilisateur veut certainement celui-ci comme avatar...
 94         if (isset($_FILES['field_string_avatarFile']) and !empty($_FILES['field_string_avatarFile']['name'])) $avatar = 'user';
 95 
 96         switch ($avatar){
 97             case 'default':   $this->user->setAvatar('default'); break;
 98             case 'gravatar':  $this->user->setAvatar('gravatar'); break;
 99             case 'user':
100                 $userAvatarFile = Sanitize::sanitizeFilename($this->user->getName()).'.png';
101                 if ((!isset($_FILES['field_string_avatarFile']) or empty($_FILES['field_string_avatarFile']['name'])) and !file_exists(AVATAR_PATH.$userAvatarFile)){
102                     // Si l'utilisateur n'a pas d'image déjà chargée et qu'il n'en a pas indiqué dans le champ adéquat, on retourne false
103                     new Alert('error', 'Impossible de sauvegarder l\'avatar, aucune image n\'a été chargée !');
104                     return false;
105                 }elseif(isset($_FILES['field_string_avatarFile']) and !empty($_FILES['field_string_avatarFile']['name'])){
106                     // Chargement de l'image
107                     $args = array();
108                     $args['resize'] = array('width' => 80, 'height' => 80);
109                     // Les avatars auront le nom des utilisateurs, et seront automatiquement transformés en .png par SimpleImages
110                     $args['name'] = $this->user->getName().'.png';
111                     $avatarFile = Upload::file($_FILES['field_string_avatarFile'], AVATAR_PATH, AVATAR_MAX_SIZE, unserialize(ALLOWED_IMAGES_EXT), $args);
112                     if (!$avatar) {
113                         new Alert('error', 'L\'avatar n\'a pas été pris en compte !');
114                         return false;
115                     }
116                     $this->user->setAvatar($avatarFile);
117                 }else{
118                     // Si l'utilisateur a déjà une image chargée et qu'il n'en a pas indiqué de nouvelle, on lui remet celle qu'il a déjà.
119                     $this->user->setAvatar($userAvatarFile);
120                 }
121                 break;
122             case 'ldap': $this->user->setAvatar('ldap');
123         }
124         // L'authentification via LDAP ramène déjà le nom l'adresse email et la gestion du mot de passe
125         if (AUTH_MODE == 'sql'){
126             if (!isset($req['name'])){
127                 new Alert('error', 'Vous n\'avez pas indiqué le nom d\'utilisateur !');
128                 return false;
129             }
130             if (!isset($req['email'])){
131                 new Alert('error', 'Vous n\'avez pas indiqué l\'adresse email !');
132                 return false;
133             }
134             $name = htmlspecialchars($req['name']);
135             if (UsersManagement::getDBUsers($name) != null and $name != $this->user->getName()){
136                 new Alert('error', 'Ce nom d\'utilisateur est déjà pris !');
137                 return false;
138             }
139             $email = $req['email'];
140             if (!\Check::isEmail($email)){
141                 new Alert('error', 'Le format de l\'adresse email que vous avez saisi est incorrect !');
142                 return false;
143             }
144             $currentPwd = (isset($req['currentPwd'])) ? $req['currentPwd'] : null;
145             $newPwd = (isset($req['newPwd'])) ? $req['newPwd'] : null;
146             if (!empty($newPwd)){
147                 // On vérifie que le mot de passe actuel a bien été saisi
148                 if (!ACL::canAdmin('admin', 0) and empty($currentPwd)){
149                     new Alert('error', 'Vous avez saisi un nouveau mot de passe sans saisir le mot de passe actuel !');
150                     return false;
151                 }
152                 // On vérifie que le mot de passe actuel est correct
153                 if (!ACL::canAdmin('admin', 0) and Login::saltPwd($currentPwd) != $this->user->getPwd()){
154                     new Alert('error', 'Le mot de passe actuel que vous avez saisi est incorrect !');
155                     return false;
156                 }
157                 // On vérifie que le nouveau mot de passe comporte bien le nombre minimum de caractères requis
158                 if (strlen($newPwd) < PWD_MIN_SIZE){
159                     new Alert('error', 'Le mot de passe doit comporter au moins '.PWD_MIN_SIZE.' caractères !');
160                     return false;
161                 }
162                 $this->user->setPwd($newPwd);
163             }
164         }
165 
166         if (UsersManagement::updateDBUser($this->user)){
167             $msg = ($this->user->getId() == $GLOBALS['cUser']->getId()) ? '' : 'de '.$this->user->getName().' ';
168           new Alert('success', 'Les paramètres du profil '.$msg.'ont été correctement sauvegardés !');
169             return true;
170         }else{
171             new Alert('error', 'Echec de la sauvegarde des paramètres !');
172             return false;
173         }
174     }

243     protected function avatarFormItems(){
244 
245         switch ($this->user->getAvatar(true)){
246             case 'default':
247             case 'ldap':
248             case 'gravatar':
249                 $valueAvatar = $this->user->getAvatar(true);
250                 break;
251             case null:
252                 $valueAvatar = 'default';
253                 break;
254             default:
255                 $valueAvatar = 'user';
256         }
257         $dataAvatar = array();
258         $dataAvatar['default'] = Avatar::display(null, 'Avatar par défaut');
259         // On propose l'avatar LDAP si celui-ci est disponible.
260         if (!empty($this->user->getLDAPProps()->avatar)) $dataAvatar['ldap'] = Avatar::display($this->user->getLDAPProps()->avatar, 'Avatar LDAP');
261         if ($this->user->getEmail() != '') $dataAvatar['gravatar'] = Avatar::display($this->user->getEmail(), 'Gravatar');
262         // On teste l'existence d'un avatar chargé par l'utilisateur
263         $userAvatarFile = Sanitize::sanitizeFilename($this->user->getName()).'.png';
264         if (file_exists(AVATAR_PATH.$userAvatarFile)){
265             $dataAvatar['user'] = Avatar::display($userAvatarFile, 'Avatar personnalisé');
266         }else{
267             $dataAvatar['user'] = 'Choisir un fichier';
268         }
269         $this->form->addField(new RadioList('avatar', $valueAvatar, 'Avatar', 'Choisissez une image pour vous représenter (avatar)', false, null, null, false, $dataAvatar, $this->user->getAvatar(true)));
270         $fileField = new File('avatarFile', null, 'Fichier de l\'avatar');
271         $fileField->setUserDefinable();
272         $this->form->addField($fileField);
