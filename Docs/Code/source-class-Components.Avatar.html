20      * retourne un avatar mis en forme
21      *
22      * @param string $avatar Avatar à afficher (dans le cas d'un gravatar, $avatar doit être une adresse email)
23      * @param string $avatarTitle Titre à mettre dans la propriété 'alt' de l'image
24      *
25      * @return string
26      */
27     public static function display($avatar = null,  $avatarTitle = 'Avatar') {
28         $pathInfo = pathinfo($avatar);
29         if (empty($avatar) or $avatar == 'default'){
30             $src = AVATAR_PATH.AVATAR_DEFAULT;
31         }elseif (isset($pathInfo['extension']) and in_array($pathInfo['extension'], array('jpg', 'jpeg', 'gif', 'png', 'bmp'))){
32             $src = AVATAR_PATH.$avatar;
33         }elseif(Check::isEmail($avatar)){
34             $src = 'http://www.gravatar.com/avatar/' . md5( strtolower( trim( $avatar ) ) ) . '?s=80';
35         }else{
36             // avatar en format binaire (venant de LDAP)
37             $src = 'data:image/jpg;base64,' . base64_encode($avatar);
38         }
39         return '<img class="img-circle avatar tooltip-bottom" alt="' . $avatarTitle . '" src="' . $src . '">';
40     }

43      * Contrôle si l'utilisateur a un gravatar ou non.
44      *
45      * @param string $email Adresse email de l'utilisateur, afin de récupérer son gravatar
46      * @return bool True si l'utilisateur a un gravatar, false sinon.'

48     public static function hasGravatar($email) {
49         // Craft a potential url and test its headers
50         $hash = md5($email);
51         $uri = 'http://www.gravatar.com/avatar/' . $hash . '?d=404';
52         $headers = @get_headers($uri);
53         if (!preg_match("|200|", $headers[0])) {
54             return FALSE;
55         } else {
56             return TRUE;
57         }
