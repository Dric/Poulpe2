 29     /**
 30      * Retourne les infos d'un utilisateur de la bdd
 31      *
 32      * Afin d'éviter de lancer une requête chaque fois qu'on veut un nom d'utilisateur, on les stocke dans une propriété statique
 33      *
 34      * @param int|string $user ID, nom ou adresse email de l'utilisateur
 35      * @param bool $nameOrEmail Si vrai, on cherche le nom ou le mail. Si faux, on ne cherche que le nom.
 36      *
 37      * @return object
 38      */
 39     static function getDBUsers($user = null, $nameOrEmail = false){
 40         global $db;
 41         if (empty(self::$usersList)) self::$usersList = $db->query('SELECT * FROM `users`');
 42         if (empty($user)){
 43             return self::$usersList;
 44         }elseif (is_numeric($user)){
 45             $userObj = \Get::getObjectsInList(self::$usersList, 'id', $user);
 46             return (isset($userObj[0])) ? $userObj[0] : false;
 47         }else{
 48             $userObj = \Get::getObjectsInList(self::$usersList, 'name', $user);
 49             if (empty($userObj) and $nameOrEmail) $userObj = \Get::getObjectsInList(self::$usersList, 'email', $user);
 50             return (isset($userObj[0])) ? $userObj[0] : false;
 51         }
 52     }

 72         global $db;

 74         if (is_numeric($user)){
 75             $where = array('id' => $user);

 77             $where = array('name' => $user);
 78         }

 85     /**
 86      * Retourne les infos d'un utilisateur LDAP
 87      *
 88      * @param string $userName Nom de l'utilisateur
 89      *
 90      * @return object|bool Informations de l'utilisateur LDAP ou false en cas d'erreur
 91      */
 92     static function getLDAPUser($userName){
 93         global $ldap;
 94         $user = $ldap->search('person', $userName, 'OU='.LDAP_AUTH_OU, array(), true);
 95         // Comme on a cherché un seul résultat, on ne veut que le premier item
 96         if ($user['count'] == 0){
 97             new Alert('error', 'Erreur : Impossible de retrouver l\'utilisateur dans l\'annuaire LDAP !');
 98             return false;
 99         }
100         $user = $user[0];
101         $LDAPUser = new LDAPUser();
102         $LDAPUser->cn = $user['cn'][0];
103         $LDAPUser->givenName = (isset($user['givenname'])) ? $user['givenname'][0] : null;
104         $LDAPUser->sn = (isset($user['sn'])) ? $user['sn'][0] : null;
105         $LDAPUser->isDisabled = ($user["useraccountcontrol"][0] == "514" or $user["useraccountcontrol"][0] == "66050") ? true : false;
106         $LDAPUser->displayName = $user['displayname'][0];
107         if (isset($user['proxyaddresses'])){
108             foreach ($user['proxyaddresses'] as $emailAdr){
109                 if (substr($emailAdr, 0, 4) == 'SMTP'){
110                     $LDAPUser->email = substr($emailAdr, 5);
111                 }
112             }
113             $LDAPUser->exchangeAlias = $user['mailnickname'][0];
114             $mdbs = explode(',', $user['homemdb'][0]);
115             $LDAPUser->exchangeBdd = ltrim($mdbs[0], 'CN=');
116         }
117         $LDAPUser->created = (isset($user['whencreated'])) ? Sanitize::ADToUnixTimestamp($user['whencreated'][0]) : null;
118         $LDAPUser->lastLogon = $ldap->lastLogon($userName);
119         $LDAPUser->groups = $ldap->userMembership($userName);
120         $LDAPUser->avatar = (isset($user['thumbnailphoto'])) ? $user['thumbnailphoto'][0] : null;
121         return $LDAPUser;
122     }

171     static function updateDBUser($User){

179         global $db;
180         $fields = array();
181         $fields['name'] = $User->getName();
182         if ($User->getEmail() != '') $fields['email'] = $User->getEmail();
183         if (method_exists($User, 'getPwd') and $User->getPwd() != null) $fields['pwd'] = $User->getPwd();
184         if ($User->getHash() != '') $fields['hash'] = $User->getHash();
185         if ($User->getAvatar(true) != '') $fields['avatar'] = $User->getAvatar(true);
186         $where = array('id'=>$User->getId());
187         return $db->update('users', $fields, $where);
