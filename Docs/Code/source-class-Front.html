 97     public static function initMainMenu(){
 98         global $cUser, $module;
 99         self::$mainMenu = new Menu('main', 'Menu');
100         // Menu vers l'administration
101         if (ACL::canAccess('admin', 0)){
102             self::$mainMenu->add(new Item('admin', 'Administration', '?module=Admin', 'Administration', null, 'menu-warning'), 98);
103         }
104         if (DISPLAY_HOME or (!DISPLAY_HOME and $module->getName() != 'home')){
105             self::$mainMenu->add(new Item('home', 'Accueil', '.', 'Revenir à l\'accueil', null, 'menu-highlight'), 2);
106         }
107         if ($cUser->isLoggedIn()){
108             self::$mainMenu->add(new Item('logoff', 'Déconnexion', '?action=logoff', 'Déconnexion de '.$cUser->getName()), 99);
109         }else{
110             self::$mainMenu->add(new Item('login', 'Connexion', '?action=loginForm', 'Authentification'), 99);
111         }
112         ModulesManagement::getModulesMenuItems();
113     }

315     /**
316      * Affiche un niveau hiérarchique dans le fil d'ariane, et ses éventuels sous-niveaux
317      *
318      * Cette fonction est récursive et permet de créer un fil d'ariane.
319      *
320      * @param  array $breadCrumb Niveau hiérarchique
321      * @param string $retLine code html déjà généré par les niveaux supérieurs
322      *
323      * @return string
324      */
325     protected static function breadCrumbLevel($breadCrumb, $retLine = ''){
326         $retLine .= '<li'.((!isset($breadCrumb['children'])) ? ' class="active"' : '').'><a href="'.$breadCrumb['link'].'">'.ucfirst($breadCrumb['title']).'</a></li>';
327         if (isset($breadCrumb['children']) and is_array($breadCrumb['children'])) return self::breadCrumbLevel($breadCrumb['children'], $retLine);
328         return $retLine;
329     }

347     /**
348      * Affiche une pagination
349      *
350      * Les url de pages sont accessibles avec `&itemsPage=<page>`, car le paramètre `&page` est utilisé par les modules pour les sous-pages
351      *
352      * @param int    $page Page actuelle
353      * @param int    $postsPerPage Nombre d'items par page
354      * @param int    $total Nombre total d'items à paginer
355      * @param string $url URL des liens
356      *
357      */
358     public static function paginate($page, $postsPerPage, $total, $url){
359         $pagesNumber = ceil($total / $postsPerPage);
360         if ($page == 0){
361             $page = 1;
362         }
363         ?>
364         <div>
365             <ul class="pagination">
366                 <li<?php    if ($page == 1){ ?> class="disabled"<?php   }   ?>>
367                     <a href="<?php echo $url; ?>&itemsPage=1">&laquo;</a>
368                 </li>
369                 <?php
370                 for ($x=1; $x<=$pagesNumber; $x++){
371                     ?>
372                 <li<?php if ($x == $page){ ?> class="active"<?php   }   ?>>
373                     <a href="<?php echo $url.'&itemsPage='.$x; ?>"><?php echo $x; ?></a>
374                 </li>
375                 <?php   }   ?>
376                 <li<?php if ($page == $pagesNumber){ ?> class="disabled"<?php   }   ?>>
377                     <a href="<?php echo $url.'&itemsPage='.($page+1); ?>">&raquo;</a>
378                 </li>
379             </ul>
380         </div>
381         <?php
382     }

384     /**
385      * Récupère la date et le hash du dernier commit local de Poulpe2
386      * @param bool $force Force la vérification
387      *
388      * @return StdClass
389      */
390     public static function getLastCommit($force = false){
391         $fs = new \FileSystem\Fs(Front::getAbsolutePath());
392         $lastCommitFile = 'cache/lastCommit.cache';
393         $heads = '.git/refs/heads';
394         if (!$fs->fileExists($lastCommitFile) or $force or $fs->getFileMeta($heads, 'dateModified')->dateModified > $fs->getFileMeta($lastCommitFile, 'dateModified')->dateModified){
395             exec('git log -1 --format="%H %h %ct"', $out);
396             $fs->writeFile($lastCommitFile, $out[0]);
397         }
398         $content = $fs->readFile($lastCommitFile, 'string');
399         $ret = new \StdClass;
400         list($ret->fullHash, $ret->hash, $ret->date) = explode(' ', $content);
401         $ret->date = (int)$ret->date;
402         return $ret;
403     }
