226     protected function saveModules(){
227         if (!ACL::canModify('admin', $this->id)){
228             new Alert('error', 'Vous n\'avez pas l\'autorisation de faire ceci !');
229             return false;
230         }
231         $modules = array();
232         foreach ($_REQUEST as $request => $value){
233             if (substr($request, 0, 7) == 'enable_'){
234                 list($dummy, $className, $type) = explode('_', $request);
235                 if (isset($modules[$className]) and $modules[$className] < $value){
236                     $modules[$className] = $value;
237                 }elseif (!isset($modules[$className])){
238                     $modules[$className] = $value;
239                 }
240             }
241         }
242         $modulesDb = ModulesManagement::getActiveModules();
243         $activatedModules = array();
244         foreach ($modulesDb as $module){
245             $activatedModules[] = $module->class;
246         }
247         foreach ($modules as $className => $value){
248             $class = 'Modules\\'.$className.'\\'.$className;
249             if (!in_array($class, $activatedModules) and $value == 1){
250                 $enableModule = new $class;
251             }elseif(in_array($class, $activatedModules) and $value == 0){
252                 if (ModulesManagement::disableModule(new $class)){
253                     new Alert('success', 'Le module a été désactivé !');
254                 }
255             }
256         }
257         ModulesManagement::getActiveModules(true);
258         return true;
259     }

404      * @use \UsersManagement::createUser()

407     protected function createUser(){
408         if (!ACL::canModify('admin', $this->id)){
409             new Alert('error', 'Vous n\'avez pas l\'autorisation de faire ceci !');
410             return false;
411         }
412         $req = $this->postedData;
413         if (!isset($req['name'])){
414             new Alert('error', 'Vous n\'avez pas indiqué le nom d\'utilisateur !');
415             return false;
416         }
417         if (!isset($req['email'])){
418             new Alert('error', 'Vous n\'avez pas indiqué l\'adresse email !');
419             return false;
420         }
421         if (!isset($req['pwd'])){
422             new Alert('error', 'Le mot de passe est vide !');
423             return false;
424         }
425         $name = htmlspecialchars($req['name']);
426         if (UsersManagement::getDBUsers($name) != null){
427             new Alert('error', 'Ce nom d\'utilisateur est déjà pris !');
428             return false;
429         }
430         $email = $req['email'];
431         if (!\Check::isEmail($email)){
432             new Alert('error', 'Le format de l\'adresse email que vous avez saisi est incorrect !');
433             return false;
434         }
435         $pwd = $req['pwd'];
436         // On vérifie que le nouveau mot de passe comporte bien le nombre minimum de caractères requis
437         if (strlen($pwd) < PWD_MIN_SIZE){
438             new Alert('error', 'Le mot de passe doit comporter au moins '.PWD_MIN_SIZE.' caractères !');
439             return false;
440         }
441         if (UsersManagement::createDBUser($name, $email, $pwd)){
442             new Alert('success', 'L\'utilisateur <code>'.$name.'</code> a été créé ! It\'s alive !');
443             return true;
444         }else{
445             new Alert('error', 'Impossible de créer l\'utilisateur <code>'.$name.'</code> !');
446             return false;
447         }

555         $readOnly = ($fileMeta->writable) ? false : true;
556         if ($readOnly){

561             }

565             if (stristr(strtolower($line), 'define')){
566                 preg_match_all('/\(.+?\)/', $line, $define); //On récupère les chaînes entre guillemets
567                 $define = explode(',', $define[0][0], 2);

572                         case 'string':
573                         default:
574                             $line = 'define(\''.$constantName.'\', \''.$req[$constantRet].'\');';
575                             break;
576                         case 'boolean':
577                             $bool = ($req[$constantRet]) ? 'true' : 'false';
578                             $line = 'define(\''.$constantName.'\', '.$bool.');';
579                             break;
580                         case 'int':
581                             $line = 'define(\''.$constantName.'\', '.$req[$constantRet].');';
582                             break;
583                         case 'array':
584                             if ($req[$constantRet]['serialize']){
585                                 array_walk($req[$constantRet]['values'], function(&$value, $key) {
586                                     $value = '\''.$value.'\'';
587                                 });
588                                 $line = 'define(\''.$constantName.'\', serialize(array('.\Sanitize::SanitizeForDb($req[$constantRet]['values'], false).')));';
589                             }else{
590                                 $line = 'define(\''.$constantName.'\', '.\Sanitize::SanitizeForDb($req[$constantRet]['values']).');';
591                             }
592                             break;
593                     }
594                 }
595             }
596         }
597         // On écrit dans le fichier
598         $ret = $share->writeFile('config.php', $configFile, false, true);
599         if (!$ret){

601             // On remet en lecture seule
602             $share->setChmod('config.php', 644);
603             return false;
604         }
605         new Alert('success', 'La modification des paramètres dans <code>config.php</code> a été prise en compte !');
606         // On remet en lecture seule
607         $share->setChmod('config.php', 644);
608         return true;
