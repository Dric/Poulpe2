 67     /**
 68      * Retourne les infos du cookie d'authentification sous forme d'objet
 69      * - $cookie->userID
 70      * - $cookie->userHash
 71      *
 72      * @return bool|Object
 73      */
 74     static function getCookie(){
 75         if (isset($_COOKIE[self::$cookieName])){
 76             $cookieArray = unserialize($_COOKIE[self::$cookieName]);
 77             $cookie = new \stdClass();
 78             $cookie->id = $cookieArray[0];
 79             $cookie->hash = $cookieArray[1];
 80             return $cookie;
 81         }
 82         return false;
 83     }

109     /**
110      * Valide la connexion d'un utilisateur
111      * @return bool
112      */
113     static function tryLogin(){
114         global $ldap;
115         $from = (isset($_REQUEST['from'])) ? $_REQUEST['from'] : '';
116         if (!isset($_REQUEST['loginName']) or empty($_REQUEST['loginName']) or !isset($_REQUEST['loginPwd']) or empty($_REQUEST['loginPwd'])) {
117             new Alert('error', 'Le nom ou le mot de passe est vide !');
118             return false;
119         }
120         $loginName = htmlspecialchars($_REQUEST['loginName']);
121         $loginPwd = htmlspecialchars($_REQUEST['loginPwd']);
122         $stayConnected = (isset($_REQUEST['stayConnected'])) ? true : false;
123         if (!empty($loginName) and !empty($loginPwd)){
124             if (AUTH_MODE == 'sql'){
125                 if ($userDb = UsersManagement::getDBUsers($loginName, true)){
126                     if ($userDb->pwd == self::saltPwd($loginPwd)) self::doLogin($loginName, $from, $stayConnected);
127                 }else{
128                     new Alert('error', 'Ce nom de connexion est inconnu !');
129                     return false;
130                 }
131             }else{
132                 if ($ldap->tryLDAPLogin($loginName, $loginPwd)){
133                     if (!UsersManagement::getDBUsers($loginName, false)){
134                         if (UsersManagement::createDBUser($loginName)){
135                             return self::doLogin($loginName, $from, $stayConnected);
136                         }else{
137                             new Alert('error', 'Impossible de créer l\'utilisateur dans la base de données !');
138                             return false;
139                         }
140                     }else{
141                         return self::doLogin($loginName, $from, $stayConnected);
142                     }
143                 }else{
144                     new Alert('debug', '<code>Login->tryLogin()</code> : Connexion LDAP échouée !');
145                 }
146             }
147         }
148         new Alert('error', 'Les identifiants sont incorrects !');
149         return false;
150     }

152     /**
153      * Effectue la connexion d'un utilisateur
154      *
155      * @param int|string $user ID ou nom de l'utilisateur
156      * @param string     $from URL vers laquelle rediriger l'utilisateur
157      * @param bool       $stayConnected Si false, le cookie expire à la fin de la session
158      *
159      * @return bool
160      */
161     static protected function doLogin($user, $from = '/', $stayConnected){
162         $cookieDuration = ($stayConnected) ? (time()+(COOKIE_DURATION*3600)) : 0;
163         $hash = UsersManagement::updateUserHash($user);
164         $ret = setcookie(COOKIE_NAME, serialize(array($user,$hash)), $cookieDuration, '/', '', FALSE, TRUE);
165         if (!$ret){
166             new Alert('error', 'Impossible de créer le cookie d\'authentification !');
167             return false;
168         }
169         if ($from != '/') {
170             $args = Get::urlParamsToArray($from);
171             unset($args['from']);
172             unset($args['action']);
173             $urlArgs = http_build_query($args);
174             header('location: index.php'. ((!empty($urlArgs)) ? '?'.$urlArgs : ''));
175         }else{
176             header('location: index.php');
177         }
178         exit();
179     }

181     /**
182      * Crée le premier utilisateur du site, si l'authentification se fait via sql.
183      *
184      * @return bool
185      */
186     protected static function createFirstUser(){
187         $req = PostedData::get();
188         if (!isset($req['name'])){
189             new Alert('error', 'Vous n\'avez pas indiqué le nom d\'utilisateur !');
190             return false;
191         }
192         if (!isset($req['email'])){
193             new Alert('error', 'Vous n\'avez pas indiqué l\'adresse email !');
194             return false;
195         }
196         if (!isset($req['pwd'])){
197             new Alert('error', 'Le mot de passe est vide !');
198             return false;
199         }
200         $name = htmlspecialchars($req['name']);
201         $email = $req['email'];
202         if (!\Check::isEmail($email)){
203             new Alert('error', 'Le format de l\'adresse email que vous avez saisi est incorrect !');
204             return false;
205         }
206         $pwd = $req['pwd'];
207         // On vérifie que le nouveau mot de passe comporte bien le nombre minimum de caractères requis
208         if (strlen($pwd) < PWD_MIN_SIZE){
209             new Alert('error', 'Le mot de passe doit comporter au moins '.PWD_MIN_SIZE.' caractères !');
210             return false;
211         }
212         if (UsersManagement::createDBUser($name, $email, $pwd)){
213             new Alert('success', 'L\'utilisateur <code>'.$name.'</code> a été créé ! It\'s alive !');
214             return true;
215         }else{
216             new Alert('error', 'Impossible de créer l\'utilisateur <code>'.$name.'</code> !');
217             return false;
218         }
219     }
