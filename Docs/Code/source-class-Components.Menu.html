 70     /**
 71      * Construction du menu
 72      *
 73      * @param string $name  Nom du menu
 74      * @param string $title Titre affiché du menu
 75      * @param string $link  Lien du menu (facultatif)
 76      * @param string $desc  Description du menu (facultatif)
 77      * @param string $icon  Icône associée au menu (facultatif)
 78      */
 79     public function __construct($name, $title, $link = '', $desc = '', $icon = ''){
 80         $this->name = Sanitize::sanitizeFilename($name);
 81         $this->title = htmlspecialchars($title);
 82         if (!empty($link)) $this->link = htmlspecialchars($link);
 83         if (!empty($desc)) $this->desc = htmlspecialchars($desc);
 84         if (!empty($icon)) $this->icon = htmlspecialchars($icon);
 85         self::$menus[$this->name] = $this;
 86     }

 90      */
 91     public function __destruct(){

 93     }

 95     /**
 96      * Ajoute un item dans le menu
 97      *
 98      * @param Item $item Item ou sous-menu de menu
 99      * @param int  $priority Priorité de l'item de menu (entre 0 et 100) (facultatif)
100      *
101      * @return bool
102      */
103     public function add(Item $item, $priority = 50){
104         // Si $item n'est ni un Item, ni un Menu, on ne l'ajoute pas.
105         if (!($item instanceof Item) and !($item instanceof Menu)){
106             new Alert('debug', '<code>Menu->add()</code> : $item n\'est ni un menu, ni un item !');
107             return false;
108         }
109         /*
110         Plusieurs items peuvent avoir la même priorité de base.
111         Pour pouvoir les traiter, on multiplie la priorité par 100,
112         ce qui permet potentiellement à 100 items d'avoir la même priorité
113         */
114 
115         $toOrder = array();
116         // On met tous les Items qui ont la même priorité que celui ajouté dans un autre tableau, et on les enlève de $this->items
117         foreach ($this->items as $itemPriority => $itemObj){
118             if ((int)($itemPriority / 100) == $priority){
119                 $toOrder[] = $itemObj;
120                 unset($this->items[$itemPriority]);
121             }
122         }
123         // N'oublions pas l'item qu'on veut ajouter au menu...
124         $toOrder[] = $item;
125         // On trie le tableau par titre
126         $toOrder = Sanitize::sortObjectList($toOrder, array('title'));
127         // On (ré)insère les items de $toOrder dans le tableau global des items
128         foreach ($toOrder as $key => $itemObj){
129             $itemKey = ($priority * 100) + $key;
130             $this->items[$itemKey] = $itemObj;
131         }
132 
133         self::$menus[$this->name]->items = $this->items;
134         return true;
135     }

143      */
144     public function build($menuClass = '', $itemClass = '', $displayTitle = false){
145         ksort ($this->items);
146         ?>
147         <div class="secondary-menu">
148             <?php if (!empty($this->title) and $displayTitle) {?><h4 class="secondary-menu-title"><?php echo $this->title; ?></h4><?php } ?>
149             <ul class="menu<?php echo (!empty($menuClass)) ? ' '.$menuClass : ''; ?>" id="menu-<?php echo $this->name; ?>">
150                 <?php
151                 foreach ($this->items as $item){
152                     /** @var Item $item */
153                     $item->build($itemClass);
154                 }
155                 ?>
156             </ul>
157         </div>
158         <?php
159     }
