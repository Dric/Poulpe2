104      * Construction du champ
105      *
106      * @param string  $name         Nom du champ - repris dans la propriété name et id
107      * @param string  $type         Type du champ - type élargi par rapport au type de Setting
108      * @param mixed   $value        Valeur à mettre dans la propriété value
109      * @param string  $label        Libellé du champ - repris dans l'élément `<label>`
110      * @param string  $placeholder  Placeholder - repris dans la propriété du même nom
111      * @param string  $help         Message d'aide du champ - facultatif
112      * @param Pattern $pattern      Validation de saisie
113      * @param bool    $important    Paramètre à signaler comme étant important
114      * @param string  $ACLLevel     Niveau minimum d'autorisation (`access`, `modify` ou `admin`)

118     public function __construct($name, $type, $value, $label = null, $placeholder = null, $help = null, Pattern $pattern = null, $important = false, $ACLLevel = 'admin', $class = '', $disabled = false){
119         self::$types = Setting::$types;
120         self::$types += array(
121             'email'   => 'string',
122           'radio'   => 'string',
123           'file'    => 'string',
124           'linkButton'  => 'string',
125           'select'  => 'string',
126           'checkboxList'  => 'array'
127         );
128         switch ($type){
129             case 'email':
130             case 'radio':
131             case 'select':
132                 $typeSetting = 'string';
133                 break;
134             default:
135                 $typeSetting = $type;
136         }
137         parent::__construct($name, $typeSetting, $value);
138 
139         $this->type = $type;
140         $this->important = (bool)$important;
141         if (!empty($label)) $this->label = $label;
142         if (!empty($placeholder)) $this->placeholder = $placeholder;
143         if (!empty($help)) $this->help = $help;
144         if (!empty($pattern)) $this->pattern = $pattern;
145         $this->ACLLevel = (!empty($ACLLevel)) ? $ACLLevel : 'admin';
146         if (!empty($class)) $this->class = $class;
147         $this->disabled = $disabled;
148     }

157     public function display($enabled = true, $userValue = false, $attrs = null){
158         /**

161         $pattern = $this->pattern;

166             if ($pattern->getMaxLength() > 0)   $displayPattern .= 'data-maxlength="'.$pattern->getMaxLength().'" ';
167             if ($pattern->getMinValue() > 0)    $displayPattern .= 'min="'.$pattern->getMinValue().'" ';
168             if ($pattern->getMaxValue() > 0)    $displayPattern .= 'max="'.$pattern->getMaxValue().'" ';
169             if ($pattern->getRegExp() !== null) $displayPattern .= 'pattern="'.$pattern->getRegExp().'" ';

179             <?php if (!empty($this->associatedIcon)) { ?>
