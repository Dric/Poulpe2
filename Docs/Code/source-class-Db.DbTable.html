134     /**
135      * Création de la table en base de données
136      *
137      * @return bool
138      */
139     public function createInDb(){
140         global $db;
141         $sql = 'CREATE TABLE IF NOT EXISTS `'.$this->name.'` (';
142         if (count($this->fields) == 0){
143             new Alert('debug', '<code>DbTable->createInDb()</code> : aucun champ n\'est défini dans le tableau d\'entrée !');
144             return false;
145         }
146         $indexes = $uniqueIndexes = $multipleIndexes = $foreignKeys = array();
147         $primaryIndex = $autoIncrementValue = null;
148         /**
149          * @var Field $field
150          */
151         foreach ($this->fields as $field){
152             $type = Sanitize::PHPToSQLType($field->getType());
153             $sql .= '`'.$field->getName().'` '.$type;
154             /**
155              * @var DbFieldSettings $settings
156              */
157             $settings = $field->getSettings();
158             if (empty($settings)){
159                 new Alert('debug', '<code>DbTable->createInDb()</code> : Les paramètres de champ <code>'.$field->getName().'</code> ne sont pas définis via un objet DbFieldSettings !');
160                 return false;
161             }
162             if ($type == 'tinyint') {
163                 $sql .= '(1)';
164             }elseif ($type != 'text' and $type != 'bool' and $settings->getLength() == 0){
165                 new Alert('debug', '<code>DbTable->createInDb()</code> : La longueur du champ <code>'.$field->getName().'</code> n\'est pas définie !');
166                 return false;
167             }elseif($type != 'bool'){
168                 $sql .= '('.$settings->getLength().')';
169             }
170             if ($settings->getRequired()){
171                 $sql .= ' NOT NULL';
172             }
173             $value = $field->getValue();
174             if (!empty($value)){
175                 if (is_string($value)) $value = str_replace('\'', ' ', $value);
176                 $sql .= ' DEFAULT \''.$value.'\'';
177             }
178             if ($field->getLabel() != null) $sql .= ' COMMENT \''.str_replace('\'', ' ', $field->getLabel()).'\'';
179             if ($settings->getAutoIncrement()){
180                 $sql .= ' AUTO_INCREMENT';
181             }
182             $sql .=', ';
183             switch ($settings->getIndex()){
184                 case 'index':
185                     $indexes[] = $field->getName();
186                     break;
187                 case 'primary':
188                     $primaryIndex = $field->getName();
189                     break;
190                 case 'unique':
191                     $uniqueIndexes[] = $field->getName();
192                     break;
193             }
194             if ($settings->getInMultipleIndex()){
195                 $multipleIndexes[] = $settings->getInMultipleIndex();
196             }
197             if ($settings->getForeignKey() !== null){
198                 $foreignKeys[$field->getName()] = $settings->getForeignKey();
199             }
200             if ($settings->getAutoIncrementValue() !== 0){
201                 $autoIncrementValue = $settings->getAutoIncrementValue();
202             }
203         }
204         if (!empty($primaryIndex)){
205             $sql .= 'PRIMARY KEY (`'.$primaryIndex.'`), ';
206         }
207         if (!empty($uniqueIndexes)){
208             foreach ($uniqueIndexes as $key){
209                 $sql .= 'UNIQUE KEY `'.$key.'` (`'.$key.'`), ';
210             }
211         }
212         if (!empty($multipleIndexes)){
213             $fieldsMultiKey = $multipleIndexes;
214             array_walk($multipleIndexes, function(&$value){
215                 $value = '`'.$value.'`';
216             });
217             $sql .= 'UNIQUE KEY `'.implode('_', $fieldsMultiKey).'` ('.implode(', ', $multipleIndexes).'), ';
218         }
219         if (!empty($indexes)){
220             foreach ($indexes as $key){
221                 $sql .= 'KEY `'.$key.'` (`'.$key.'`), ';
222             }
223         }
224         if (!empty($foreignKeys)){
225             /**
226              * @var ForeignKey $foreignKey
227              */
228             foreach ($foreignKeys as $key => $foreignKey){
229                 $sql .= 'FOREIGN KEY (`'.$key.'`) REFERENCES `'.$foreignKey->getTable().'` (`'.$foreignKey->getKey().'`) ON DELETE '.$foreignKey->getOnDelete().' ON UPDATE '.$foreignKey->getOnUpdate().', ';
230             }
231         }
232         $sql = rtrim($sql, ', ');
233         $sql .= ') ENGINE=InnoDB DEFAULT CHARSET=utf8';
234         if (!empty($autoIncrementValue)){
235             $sql .= ' AUTO_INCREMENT='.$autoIncrementValue;
236         }
237         $sql .= ';';
238         $ret = $db->query($sql);
239         return ($ret !== false) ? true : false;
240     }

242     /**
243      * Insère des lignes dans la table
244      *
245      * Les valeurs à insérer sont définies en tant que tableau associatif :
246      *
247      *  $row[ID] = array(
248      *    champ   => Valeur
249      *    champ2  => Valeur2
250      *  );
251      *
252      * @param array $rows Tableau associatif de lignes à insérer, avec l'ID de la ligne en tant que clé
253      *
254      * @return bool
255      */
256     public function insertRows($rows){
257         global $db;
258         if (!empty($rows)){
259             $itemsIdsDb = $db->get($this->name, 'id');
260 
261             $itemsToDelete = $keysToUpdate = array();
262             foreach ($itemsIdsDb as $itemId){
263                 $itemsToDelete[] = $itemId->id;
264             }
265             $sql = 'INSERT INTO `'.$this->name.'` (';
266             /**
267              * @var Field $field
268              */
269             foreach ($this->fields as $field){
270                 $sql .= '`'.$field->getName().'`, ';
271                 if ($field->getSettings()->getOnDuplicateKeyUpdate()){
272                     $keysToUpdate[] = $field->getName();
273                 }
274             }
275             $sql = rtrim($sql, ', ').') VALUES ';
276             // Les valeurs à présent
277             foreach ($rows as $id => $row){
278                 $sql .= '(';
279                 // Si l'ID n'est pas affichée (et donc pas renvoyée par le formulaire), il va falloir la renseigner quand même via le nom du champ renvoyé
280                 if (!isset($row['id'])){
281                     if ($id != 'new' or empty($id)){
282                         $sql .= $id.', ';
283                         // On enlève l'id du tableau des ids d'items, afin de pouvoir supprimer les lignes restantes
284                         unset($itemsToDelete[array_search($id, $itemsToDelete)]);
285                     }else{
286                         // Si l'ID est 'new', c'est un nouveau champ. Pour laisser faire l'auto-incrémentation, on envoie une valeur nulle
287                         $sql .= 'NULL, ';
288                     }
289                 }
290                 // Valeurs de chaque colonne pour une ligne
291                 foreach ($this->fields as $field){
292                     // On vient de traiter l'id juste avant la boucle, on l'enlève donc des champs à traiter
293                     if ($field->getName() != 'id'){
294                         $sql .= ((isset($row[$field->getName()])) ? '"'.str_replace('\\', '\\\\', $row[$field->getName()]).'"' : 'NULL').', ';
295                     }
296                 }
297                 $sql = rtrim($sql, ', ');
298                 $sql .= '),';
299             }
300             $sql = rtrim($sql, ', ');
301 
302             // Évidemment, on risque de trouver des enregistrements déjà présents. Grâce à 'onDuplicateKeyUpdate', on va savoir quels colonnes mettre à jour en cas d'enregistrements déjà présents
303             if (!empty($keysToUpdate)){
304                 $sql .= ' ON DUPLICATE KEY UPDATE ';
305                 foreach ($keysToUpdate as $update){
306                     $sql.= '`'.$update.'` = VALUES(`'.$update.'`), ';
307                 }
308                 $sql = rtrim($sql, ', ');
309             }
310             // Enfin, on exécute la requête SQL.
311             $ret = $db->query($sql);
312             // On supprime les enregistrements qui n'ont pas été renvoyés par le formulaire, ce qui signifie qu'ils ont été effacés.
313             $ret2 = true;
314             if (!empty($itemsToDelete)){
315                 foreach ($itemsToDelete as $item){
316                     $ret2 = $db->delete($this->name, array('id'=>$item));
317                 }
318             }
319             if ($ret === false or $ret2 === false){
320                 new Alert('error', 'Impossible de sauvegarder les enregistrements de la table <code>'.$this->name.'</code> !');
321                 return false;
322             }else{
323                 new Alert('success', 'Les enregistrements de la table <code>'.$this->name.'</code> ont été mis à jour.');
324                 return true;
325             }
326         }else{
327             return true;
328         }
