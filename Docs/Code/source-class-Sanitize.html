 20     /**
 21      * Supprime tout caractère louche, accentué ou autre d'une chaîne
 22      *
 23      * @from <www.house6.com/blog/?p=83>
 24      * @param string $f Chaîne à traiter
 25      * @param bool $toLower Retourner la chaîne en minuscules
 26      *
 27      * @return string
 28      */
 29     static function sanitizeFilename($f, $toLower = true) {
 30         // a combination of various methods
 31         // we don't want to convert html entities, or do any url encoding
 32         // we want to retain the "essence" of the original file name, if possible
 33         // char replace table found at:
 34         // http://www.php.net/manual/en/function.strtr.php#98669
 35         $replace_chars = array(
 36             'Š'=>'S', 'š'=>'s', 'Ð'=>'Dj','Ž'=>'Z', 'ž'=>'z', 'À'=>'A', 'Á'=>'A', 'Â'=>'A', 'Ã'=>'A', 'Ä'=>'A',
 37             'Å'=>'A', 'Æ'=>'A', 'Ç'=>'C', 'È'=>'E', 'É'=>'E', 'Ê'=>'E', 'Ë'=>'E', 'Ì'=>'I', 'Í'=>'I', 'Î'=>'I',
 38             'Ï'=>'I', 'Ñ'=>'N', 'Ò'=>'O', 'Ó'=>'O', 'Ô'=>'O', 'Õ'=>'O', 'Ö'=>'O', 'Ø'=>'O', 'Ù'=>'U', 'Ú'=>'U',
 39             'Û'=>'U', 'Ü'=>'U', 'Ý'=>'Y', 'Þ'=>'B', 'ß'=>'Ss','à'=>'a', 'á'=>'a', 'â'=>'a', 'ã'=>'a', 'ä'=>'a',
 40             'å'=>'a', 'æ'=>'a', 'ç'=>'c', 'è'=>'e', 'é'=>'e', 'ê'=>'e', 'ë'=>'e', 'ì'=>'i', 'í'=>'i', 'î'=>'i',
 41             'ï'=>'i', 'ð'=>'o', 'ñ'=>'n', 'ò'=>'o', 'ó'=>'o', 'ô'=>'o', 'õ'=>'o', 'ö'=>'o', 'ø'=>'o', 'ù'=>'u',
 42             'ú'=>'u', 'û'=>'u', 'ý'=>'y', 'þ'=>'b', 'ÿ'=>'y', 'ƒ'=>'f'
 43         );
 44         $f = strtr($f, $replace_chars);
 45         // convert & to "and", @ to "at", and # to "number"
 46         $f = preg_replace(array('/[\&]/', '/[\@]/', '/[\#]/'), array('-and-', '-at-', '-number-'), $f);
 47         $f = preg_replace('/[^(\x20-\x7F)]*/','', $f); // removes any special chars we missed
 48         $f = str_replace(' ', '-', $f); // convert space to hyphen
 49         $f = str_replace('\'', '', $f); // removes apostrophes
 50         $f = preg_replace('/[^\w\-\.]+/', '', $f); // remove non-word chars (leaving hyphens and periods)
 51         $f = preg_replace('/[\-]+/', '-', $f); // converts groups of hyphens into one
 52         if ($toLower)   return strtolower($f);
 53         return $f;
 54         }

 56     /**
 57      * Transforme un tableau en chaîne de caractères
 58      * (typiquement pour une sauvegarde en bdd)
 59      *
 60      * Si le tableau est associatif, alors il sera sérialisé pour conserver les clés.
 61      *
 62      * @param array  $array Tableau
 63      * @param string $glue Chaîne de liaison entre les valeurs
 64      *
 65      * @param bool|string $sanitizeFor Met en forme les valeurs du tableau. Peut prendre les valeurs suivantes :
 66      *  - false (défaut)
 67      *  - `db`
 68      *  - `js`
 69      *  - `file`
 70      *
 71      * @return string
 72      */
 73     public static function arrayToString(array $array, $glue = ', ', $sanitizeFor = false){
 74         if ($sanitizeFor !== false){
 75             if ($sanitizeFor == 'db'){
 76                 foreach ($array as &$value){
 77                     $value = self::SanitizeForDb($value, false);
 78                 }
 79             }
 80             if ($sanitizeFor == 'js'){
 81                 foreach ($array as &$value){
 82                     $value = self::SanitizeForJs($value);
 83                 }
 84             }
 85             if ($sanitizeFor == 'file'){
 86                 foreach ($array as &$value){
 87                     $value = self::sanitizeFilename($value);
 88                 }
 89             }
 90         }
 91         if (Check::isAssoc($array)){
 92             return serialize($array);
 93         }else{
 94             return implode($glue, $array);
 95         }
 96     }

 98     /**
 99      * Transforme un format de date en un autre
100      *
101      * Cette méthode accepte soit un timestamp soit une date normale en entrée
102      *
103      * @param int|string $date Date à transformer
104      * @param string $to Format de retour
105      * - timestamp
106      * - date française
107      * - dateTime française (date et heure)
108      * - fullDateTime française (date, heure, minutes, secondes)
109      * - dateAtTime française (date et heure du type 24/12/2014 à 08h45)
110      * - time Heure, minutes et secondes
111      *
112      * @return bool|int|string|null
113      */
114     public static function date($date, $to = 'timestamp'){
115         date_default_timezone_set('Europe/Paris');
116         if (empty($date)) return null;
117         switch ($to){
118             case 'timestamp':
119                 // Au cas où la date soit déjà un timestamp
120                 if (is_numeric($date)) return $date;
121                 return strtotime($date);
122             case 'date' :       return date('d/m/Y', $date);
123             case 'dateTime':    return date('d/m/Y H:i', $date);
124             case 'fullDateTime':    return date('d/m/Y H:i:s', $date);
125             case 'dateAtTime':  return date('d/m/Y à H:i', $date);
126             case 'time' :       return date('H:i:s', $date);
127             default:            return false;
128         }
129     }

145         if (empty($time)) return null;
146         switch ($to){
147             case 'timestamp':
148                 // Au cas où l'heure soit déjà un timestamp
149                 if (is_numeric($time)) return $time;
150                 $parsed = date_parse($time);
151                 return (int)($parsed['hour'] * 3600 + $parsed['minute'] * 60 + $parsed['second']);
152             case 'time' :       return gmdate('H:i', $time);
153             case 'fullTime':    return gmdate('H:i:s', $time);
154             default:            return false;

158     /**
159      * Transforme unee durée en secondes en durée lisible par l'être humain
160      *
161      * @param int $time Durée en secondes
162      * @return string
163      */
164     public static function timeDuration($time){
165         $s = $time % 60;
166         $m = (floor(($time%3600)/60)>0)?floor(($time%3600)/60) : '';
167         if (!empty($m)) $m .= ' minute'.(($m != 1) ? 's' : '');
168         $h = (floor(($time % 86400) / 3600)>0)?floor(($time % 86400) / 3600) : '';
169         if (!empty($h)) $h .= ' heure'.(($h != 1) ? 's' : '');
170         $d = (floor(($time % 2592000) / 86400)>0)?floor(($time % 2592000) / 86400) : '';
171         if (!empty($d)) $d .= ' jour'.(($d != 1) ? 's' : '');
172         $M = (floor($time / 2592000)>0)?floor($time / 2592000).' mois' : '';
173         return $M.' '.$d.' '.$h.' '.$m.' et '.$s.' seconde'.(($s != 1) ? 's': '');
174     }

238     public static function SanitizeForDb($value, $quotes = true){
239         if (is_array($value)){
240             $value = Sanitize::arrayToString($value, ', ', 'db');
241         }elseif (is_bool($value)){
242             $value = ($value) ? 1 : 0;
243         }else{
244             $value = str_replace('\\', '\\\\', $value);
245             $value = htmlspecialchars($value);
246         }
247         if (!is_numeric($value) and $quotes) $value = '\''.$value.'\'';
248         return $value;

263     /**
264      * Transforme un timestamp Active Directory en timestamp unix
265      *
266      * @see <http://stackoverflow.com/a/4647445/1749967>
267      * @see <http://maxvit.net/convert_ldap_dates>
268      * @param string $date Timestamp Active directory
269      * @return int
270      */
271     public static function ADToUnixTimestamp($date){
272         if (strpos($date, '.') === false){
273             $win_secs = substr($date,0,strlen($date)-7); // divide by 10 000 000 to get seconds
274             return ($win_secs - 11644473600); // 1.1.1600 -> 1.1.1970 difference in seconds
275         }else{
276             // La date dans l'AD est déjà réglée selon le fuseau horaire dans lequel se trouve le serveur, pour obtenir un timestamp correct, il faut remettre tricher avec les fuseaux horaires
277             date_default_timezone_set('UTC');
278             $tab = explode('.', $date);
279             list($millenium, $year, $month, $day, $hours, $minutes, $seconds) = str_split($tab[0], 2);
280             $time = mktime($hours, $minutes, $seconds, $month, $day, $millenium.$year);
281             date_default_timezone_set('Europe/Paris');
282             return $time;
283         }
284     }

286     /**
287      * Converti une valeur en octets en taille lisible (Mo, Go, etc.)
288      *
289      * @param int $size Taille en octets
290      * @param int $NbDecimals Nombre de décimales après la virgule
291      *
292      * @return string
293      */
294     public static function readableFileSize($size, $NbDecimals = 2){
295         if ($size == 0) return 'Vide';
296         $siPrefix = array( 'o', 'Ko', 'Mo', 'Go', 'To', 'Eo', 'Zo', 'Yo' );
297         $base = log(floatval($size)) / log(1024);
298         return round(pow(1024, $base - floor($base)), $NbDecimals) . $siPrefix[floor($base)];
299     }
