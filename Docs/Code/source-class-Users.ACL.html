 43      * Retourne les autorisations d'un utilisateur
 44      *
 45      * @param int $user ID d'un utilisateur
 46      *
 47      * @return bool|array
 48      */
 49     public static function getUserACL($user){
 50         global $db;
 51         if (!is_int($user)){
 52             new Alert('debug', '<code>ACL::getUserACL()</code> : <code>$user</code> n\'est pas un nombre entier ! '. Get::varDump($user));
 53             return false;
 54         }
 55         $dbACL = $db->get('ACL', '*', array('user'=>$user));
 56         if ($dbACL === false) return false;
 57         $retACL = array();
 58         foreach ($dbACL as $acl){
 59             $acl->value = ($acl->value == 1) ? true : false;
 60             $retACL[$acl->component][$acl->id][$acl->type] = $acl->value;
 61         }
 62         $dbACL = $db->get('ACL', '*', array('user'=>10000));
 63         if ($dbACL === false) return false;
 64         foreach ($dbACL as $acl){
 65             $acl->value = ($acl->value == 1) ? true : false;
 66             if (!isset($retACL[$acl->component][$acl->id][$acl->type])) $retACL[$acl->component][$acl->id][$acl->type] = $acl->value;
 67         }
 68         return $retACL;
 69     }

110     /**
111      * Détermine si l'utilisateur a le droit d'effectuer une action $type sur un composant
112      * @param string $component Composant (module, site,...)
113      * @param int $componentId Id du composant
114      * @param int $userId Id de l'utilisateur (utilisateur courant si vide)
115      * @param string $type Type d'ACL (admin, access)
116      *
117      * @return bool
118      */
119     public static function can($component, $componentId, $userId = null, $type){
120         global $db, $cUser;
121         if (!in_array($type, array_keys(self::$typesLabels))){
122             New Alert('debug', '<code>ACL::can()</code> : <code>$type='.$type.'</code> ne fait pas partie des types autorisés !');
123             return false;
124         }
125         if (empty($userId) or $userId == $cUser->getId()){
126             // Il va de soi qu'on refuse les utilisateurs non connectés si l'authentification est obligatoire.
127             if (!$cUser->isLoggedIn() and AUTH_MANDATORY) return false;
128             // On demande donc une autorisation pour l'utilisateur courant. Comme on a déjà la liste de ses autorisations, on va la chercher dans son objet
129             $userACL = $cUser->getACL();
130             return (isset($userACL[$component][$componentId][$type])) ? $userACL[$component][$componentId][$type] : false;
131         }else{
132             $where = array(
133                 'component' => $component,
134               'id'        => $componentId,
135               'user'      => $userId,
136               'type'      => $type
137             );
138             $ret = $db->getVal('ACL', 'value', $where);
139             switch ($ret){
140                 case null:
141                     // On regarde s'il existe une valeur pour tous les utilisateurs (10000 = tout le monde)
142                     $where['user'] = 10000;
143                     $ret = $db->getVal('ACL', 'value', $where);
144                     if (!is_null($ret) or $ret) return true;
145                     break;
146                 case 0:
147                 case false:
148                     return false;
149                 case 1:
150                 case true:
151                     return true;
152             }
153         }
154         return false;
155     }

157     /**
158      * Définit une règle $type sur un composant
159      *
160      * @param string $component Composant (module, site,...)
161      * @param int    $componentId Id du composant
162      * @param int    $userId Id de l'utilisateur
163      * @param string $type Type d'ACL (admin, access)
164      * @param bool   $value Valeur de l'ACL (true : permis, false : refusé)
165      *
166      * @return bool
167      */
168     public static function set($component, $componentId, $userId, $type, $value){
169         global $db;
170         if (!in_array($type, array_keys(self::$typesLabels))){
171             New Alert('debug', '<code>ACL::set()</code> : <code>$type='.$type.'</code> ne fait pas partie des typesLabels autorisés !');
172             return false;
173         }
174         $value = Sanitize::SanitizeForDb($value);
175         $types = array_flip(array_keys(self::$typesLabels));
176         $ret = false;
177         $sql = '';
178         if ($value == 1){
179             // On inscrit aussi les permissions inférieures si c'est une autorisation
180             for ($i = 0; $i <= $types[$type]; $i++){
181                 $typeI = array_search($i, $types);
182                 $sql = 'INSERT INTO `ACL` (`component`, `id`, `user`, `type`, `value`) VALUES ("'.$component.'", '.$componentId.', '.$userId.', "'.$typeI.'", '.$value.') ON DUPLICATE KEY UPDATE value = '.$value;
183                 $ret = $db->query($sql);
184                 if ($ret === false){
185                     new Alert('error', 'Impossible de sauvegarder les permissions !<br>'.$sql);
186                     return false;
187                 }
188             }
189         }else{
190             // On inscrit aussi les permissions supérieures si c'est un refus
191             for ($i = 2; $i > $types[$type]; $i--){
192                 $typeI = array_search($i, $types);
193                 $sql = 'INSERT INTO `ACL` (`component`, `id`, `user`, `type`, `value`) VALUES ("'.$component.'", '.$componentId.', '.$userId.', "'.$typeI.'", '.$value.') ON DUPLICATE KEY UPDATE value = '.$value;
194                 $ret = $db->query($sql);
195                 if ($ret === false){
196                     new Alert('error', 'Impossible de sauvegarder les permissions !<br>'.$sql);
197                     return false;
198                 }
199             }
200         }
201         return true;
202     }

204     /**
205      * Définit des règles $type sur des composants
206      *
207      * Cette fonction fait la même chose que ACL::set(), mais avec un tableau d'ACL.
208      * @param array $ACLArr Tableau d'ACL de la forme $array[(string)$component][(int)$componentId][(int)$user][(string)$type] = (bool)$value
209      *
210      * @return bool
211      */
212     public static function setArray(array $ACLArr){
213         global $db;
214         $types = array_flip(array_keys(self::$typesLabels));
215         $setACLArray = array();
216         foreach ($ACLArr as $component => $ACLComponent){
217             foreach ($ACLComponent as $componentId => $ACLCompId){
218                 foreach ($ACLCompId as $user => $ACLUser){
219                     /*
220                      * Si on donne l'autorisation d'admin sur une ressource, il faut également autoriser les accès plus restrictifs.
221                      * Ex : Si on donne le droit d'admin sur une ressource, il faut également accorder les droits access et modif.
222                      */
223                     $max = 0;
224                     $valueMax = 0;
225                     foreach ($ACLUser as $type => $value){
226                         if ($types[$type] >= $max and $value === true) {
227                             $valueMax = Sanitize::SanitizeForDb($value);
228                             $max = $types[$type];
229                         }
230                     }
231                     foreach ($ACLUser as $type => $value){
232                         $setACLArray[] = '("'.$component.'", '.$componentId.', '.$user.', "'.$type.'", '.(($types[$type] <= $max) ? $valueMax : Sanitize::SanitizeForDb($value)).')';
233                     }
234                 }
235             }
236         }
237         $ret = $db->query('INSERT INTO `ACL` (`component`, `id`, `user`, `type`, `value`) VALUES '.implode(', ', $setACLArray).' ON DUPLICATE KEY UPDATE `value` = VALUES(`value`)');
238         return ( $ret === false) ? false : true;
239     }

241     /**
242      * Récupère l'envoi d'un formulaire pour sauvegarder des ACL
243      *
244      * @param string $action Action à traiter.
245      *  - `restoreDefault` : restaure les droits d'accès par défaut pour un utilisateur
246      *
247      * @return bool
248      */
249     public static function requestACLSave($action = null){
250         switch ($action){
251             case 'restoreDefault':
252                 $ret = PostedData::checkToken('adminACL', $_REQUEST['token']);
253                 if (!$ret){
254                     new Alert('error', 'Erreur de traitement : Ce formulaire a déjà été envoyé, ou bien vous n\'êtes pas la personne qui a initié l\'envoi !');
255                     return false;
256                 }
257                 if (!self::canAdmin('admin', 0)){
258                     new Alert('error', 'Vous n\'avez pas l\'autorisation de faire ceci !');
259                     return false;
260                 }
261                 $user = (int)$_REQUEST['user'];
262                 if (!UsersManagement::getDBUsers($user)){
263                     new Alert('error', 'Cet utilisateur n\'existe pas !');
264                     return false;
265                 }
266                 return self::delete(null, null, $user);
267                 break;
268             case 'updateACL':
269             default:
270                 $ret = PostedData::checkToken('adminACL', $_REQUEST['token']);
271                 if (!$ret){
272                     new Alert('error', 'Erreur de traitement : Ce formulaire a déjà été envoyé, ou bien vous n\'êtes pas la personne qui a initié l\'envoi !');
273                     return false;
274                 }
275                 $ACLArr = array();
276                 foreach ($_REQUEST as $key => $value){
277                     if (substr($key, 0, 4) == 'ACL_'){
278                         list($dummy, $component, $componentId, $user, $type, $field) = explode('_', $key);
279                         if (!self::canAdmin($component, $componentId)){
280                             new Alert('error', 'Vous n\'avez pas l\'autorisation de faire ceci !');
281                             return false;
282                         }
283                         if (!in_array($type, array_keys(self::$typesLabels))){
284                             New Alert('debug', '<code>ACL::requestACLSave()</code> : <code>$type='.$type.'</code> ne fait pas partie des typesLabels autorisés !');
285                             return false;
286                         }
287                         $ACLArr[$component][$componentId][$user][$type][$field] = ($value == 1) ? true : false;
288                     }
289                 }
290                 foreach ($ACLArr as $component => $ACLArrComponent) {
291                     foreach ($ACLArrComponent as $componentId => $ACLArrComponentId) {
292                         foreach ($ACLArrComponentId as $user => $ACLUser) {
293                             foreach ($ACLUser as $type => $ACLType) {
294                                 $ACLArr[$component][$componentId][$user][$type] = (isset($ACLType['checkbox'])) ? $ACLType['checkbox'] : $ACLType['hidden'];
295                             }
296                         }
297                     }
298                 }
299                 if ($ret = self::setArray($ACLArr)){
300                     new Alert('success', 'Les autorisations ont été sauvegardées !');
301                 }else{
302                     new Alert('error', 'Les autorisations n\'ont pas été sauvegardées !');
303                 }
304                 return $ret;
305         }
306     }

308     /**
309      * Supprime des ACL
310      *
311      * Si $component est vide, alors on supprime tout (si $type est vide) ou une partie (si $type est renseigné) des ACL d'un utilisateur
312      * Si $userId est vide, on réinitialise les ACL pour un composant (ou une partie si $type est renseigné)
313      *
314      * @param string $component Composant (module, site,...)
315      * @param int $componentId Id du composant
316      * @param int $userId Id de l'utilisateur
317      * @param string $type Type d'ACL (admin, access)
318      *
319      * @return bool
320      */
321     public static function delete($component = null, $componentId = null, $userId = null, $type = null){
322         global $db;
323         $where = array();
324         $format = array(
325             'component' => 'string',
326           'id'        => 'int',
327           'user'      => 'int',
328           'type'      => 'string'
329         );
330         if (!is_null($component)){
331             if (is_null($componentId)) {
332                 new Alert('debug', '<code>ACL::delete()</code> : $componentId est vide alors que $component est renseigné !');
333                 return false;
334             }
335             $where['component'] = $component;
336             $where['id'] = $componentId;
337         }
338         if (!is_null($userId)){
339             $where['user'] = $userId;
340         }
341         if (!is_null($type)){
342             if (!in_array($type, array_keys(self::$typesLabels))){
343                 New Alert('debug', '<code>ACL::delete()</code> : <code>$type='.$type.'</code> ne fait pas partie des typesLabels autorisés !');
344                 return false;
345             }
346             $where['type'] = $type;
347         }
348         if (empty($where) or (isset($where['type']) and count($where) == 1)){
349             // Si aucun paramètre (ou seulement le type) n'est renseigné, on annule tout
350             new Alert('debug', '<code>ACL::delete()</code> : Aucun paramètre n\'est renseigné !');
351             return false;
352         }
353         $ret = $db->delete('ACL', $where, $format);
354         if ($ret === false){
355             new Alert('error', 'Impossible de supprimer les autorisations !');
356             return false;
357         }
358         new Alert('success', 'Les autorisations ont été correctement supprimées !');
359         return true;
360     }
