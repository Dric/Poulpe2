 43      *
 44      * @param int $user ID d'un utilisateur

 46      * @return bool|array
 47      */
 48     public static function getUserACL($user){
 49         global $db;
 50         if (!is_int($user)){
 51             new Alert('debug', '<code>ACL::getUserACL()</code> : <code>$user</code> n\'est pas un nombre entier ! '. Get::varDump($user));
 52             return false;
 53         }
 54         $dbACL = $db->get('ACL', '*', array('user'=>$user));
 55         if ($dbACL === false) return false;
 56         $retACL = array();
 57         foreach ($dbACL as $acl){
 58             $acl->value = ($acl->value == 1) ? true : false;
 59             $retACL[$acl->component][$acl->id][$acl->type] = $acl->value;
 60         }
 61         $dbACL = $db->get('ACL', '*', array('user'=>10000));
 62         if ($dbACL === false) return false;
 63         foreach ($dbACL as $acl){
 64             $acl->value = ($acl->value == 1) ? true : false;
 65             if (!isset($retACL[$acl->component][$acl->id][$acl->type])) $retACL[$acl->component][$acl->id][$acl->type] = $acl->value;
 66         }
 67         return $retACL;
 68     }

109     /**
110      * Détermine si l'utilisateur a le droit d'effectuer une action $type sur un composant
111      * @param string $component Composant (module, site,...)
112      * @param int $componentId Id du composant
113      * @param int $userId Id de l'utilisateur (utilisateur courant si vide)
114      * @param string $type Type d'ACL (admin, access)
115      *
116      * @return bool
117      */
118     public static function can($component, $componentId, $userId = null, $type){
119         global $db, $cUser;
120         if (!in_array($type, array_keys(self::$typesLabels))){
121             New Alert('debug', '<code>ACL::can()</code> : <code>$type='.$type.'</code> ne fait pas partie des types autorisés !');
122             return false;
123         }
124         if (empty($userId) or $userId == $cUser->getId()){
125             // On demande donc une autorisation pour l'utilisateur courant. Comme on a déjà la liste de ses autorisations, on va la chercher dans son objet
126             $userACL = $cUser->getACL();
127             return (isset($userACL[$component][$componentId][$type])) ? $userACL[$component][$componentId][$type] : false;
128         }else{
129             $where = array(
130                 'component' => $component,
131               'id'        => $componentId,
132               'user'      => $userId,
133               'type'      => $type
134             );
135             $ret = $db->getVal('ACL', 'value', $where);
136             switch ($ret){
137                 case null:
138                     // On regarde s'il existe une valeur pour tous les utilisateurs (10000 = tout le monde)
139                     $where['user'] = 10000;
140                     $ret = $db->getVal('ACL', 'value', $where);
141                     if (!is_null($ret) or $ret) return true;
142                     break;
143                 case 0:
144                 case false:
145                     return false;
146                 case 1:
147                 case true:
148                     return true;
149             }
150         }
151         return false;
152     }

175         if ($value == 1){

183                     return false;
184                 }
185             }
186         }else{
187             // On inscrit aussi les permissions supérieures si c'est un refus
188             for ($i = 2; $i > $types[$type]; $i--){

201     /**
202      * Définit des règles $type sur des composants
203      *
204      * Cette fonction fait la même chose que ACL::set(), mais avec un tableau d'ACL.
205      * @param array $ACLArr Tableau d'ACL de la forme $array[(string)$component][(int)$componentId][(int)$user][(string)$type] = (bool)$value
206      *
207      * @return bool
208      */
209     public static function setArray(array $ACLArr){
210         global $db;
211         $types = array_flip(array_keys(self::$typesLabels));
212         $setACLArray = array();
213         foreach ($ACLArr as $component => $ACLComponent){
214             foreach ($ACLComponent as $componentId => $ACLCompId){
215                 foreach ($ACLCompId as $user => $ACLUser){
216                     /*
217                      * Si on donne l'autorisation d'admin sur une ressource, il faut également autoriser les accès plus restrictifs.
218                      * Ex : Si on donne le droit d'admin sur une ressource, il faut également accorder les droits access et modif.
219                      */
220                     $max = 0;
221                     $valueMax = 0;
222                     foreach ($ACLUser as $type => $value){
223                         if ($types[$type] >= $max and $value === true) {
224                             $valueMax = Sanitize::SanitizeForDb($value);
225                             $max = $types[$type];
226                         }
227                     }
228                     foreach ($ACLUser as $type => $value){
229                         $setACLArray[] = '("'.$component.'", '.$componentId.', '.$user.', "'.$type.'", '.(($types[$type] <= $max) ? $valueMax : Sanitize::SanitizeForDb($value)).')';
230                     }
231                 }
232             }
233         }
234         $ret = $db->query('INSERT INTO `ACL` (`component`, `id`, `user`, `type`, `value`) VALUES '.implode(', ', $setACLArray).' ON DUPLICATE KEY UPDATE `value` = VALUES(`value`)');
235         return ( $ret === false) ? false : true;
236     }

238     /**
239      * Récupère l'envoi d'un formulaire pour sauvegarder des ACL
240      * @return bool
241      */
242     public static function requestACLSave(){
243         $ACLArr = array();
244         foreach ($_REQUEST as $key => $value){
245             if (substr($key, 0, 4) == 'ACL_'){
246                 list($dummy, $component, $componentId, $user, $type, $field) = explode('_', $key);
247                 if (!in_array($type, array_keys(self::$typesLabels))){
248                     New Alert('debug', '<code>ACL::requestACLSave()</code> : <code>$type='.$type.'</code> ne fait pas partie des typesLabels autorisés !');
249                     return false;
250                 }
251                 $ACLArr[$component][$componentId][$user][$type][$field] = ($value == 1) ? true : false;
252             }
253         }
254         foreach ($ACLArr[$component][$componentId] as $user => $ACLUser){
255             foreach ($ACLUser as $type => $ACLType){
256                 $ACLArr[$component][$componentId][$user][$type] = (isset($ACLType['checkbox'])) ? $ACLType['checkbox'] : $ACLType['hidden'];
257             }
258         }
259         if ($ret = self::setArray($ACLArr)){
260             new Alert('success', 'Les autorisations ont été sauvegardées !');
261         }else{
262             new Alert('error', 'Les autorisations n\'ont pas été sauvegardées !');
263         }
264         return $ret;
265     }

267     /**
268      * Supprime des ACL
269      *
270      * Si $component est vide, alors on supprime tout (si $type est vide) ou une partie (si $type est renseigné) des ACL d'un utilisateur
271      * Si $userId est vide, on réinitialise les ACL pour un composant (ou une partie si $type est renseigné)
272      *
273      * @param string $component Composant (module, site,...)
274      * @param int $componentId Id du composant
275      * @param int $userId Id de l'utilisateur
276      * @param string $type Type d'ACL (admin, access)
277      *
278      * @return bool
279      */
280     public static function delete($component = null, $componentId = null, $userId = null, $type = null){
281         global $db;
282         $where = array();
283         $format = array(
284             'component' => 'string',
285           'id'        => 'int',
286           'user'      => 'int',
287           'type'      => 'string'
288         );
289         if (!is_null($component)){
290             if (is_null($componentId)) {
291                 new Alert('debug', '<code>ACL::delete()</code> : $componentId est vide alors que $component est renseigné !');
292                 return false;
293             }
294             $where['component'] = $component;
295             $where['id'] = $componentId;
296         }
297         if (!is_null($userId)){
298             $where['user'] = $userId;
299         }
300         if (!is_null($type)){
301             if (!in_array($type, array_keys(self::$typesLabels))){
302                 New Alert('debug', '<code>ACL::delete()</code> : <code>$type='.$type.'</code> ne fait pas partie des typesLabels autorisés !');
303                 return false;
304             }
305             $where['type'] = $type;
306         }
307         if (empty($where) or (isset($where['type']) and count($where) == 1)){
308             // Si aucun paramètre (ou seulement le type) n'est renseigné, on annule tout
309             new Alert('debug', '<code>ACL::delete()</code> : Aucun paramètre n\'est renseigné !');
310             return false;
311         }
312         if ($ret = $db->delete('ACL', $where, $format)){
313             new Alert('error', 'Impossible de supprimer les autorisations !');
314             return false;
315         }
316         new Alert('success', 'Les autorisations ont été correctement supprimées !');
317         return true;
318     }
